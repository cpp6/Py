class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def Insert_End(self, data):
        temp = Node(data)
        if self.head is None:
            self.head = temp
        else:
            q = self.head
            while q.next:
                q = q.next
            q.next = temp

    def Display(self):
        q = self.head
        while q:
            print(q.data, end="->")
            q = q.next
        print("None")

    def union(self, other):
        s = set()
        q = self.head
        while q:
            s.add(q.data)
            q = q.next
        q = other.head
        while q:
            s.add(q.data)
            q = q.next

        u = LinkedList()
        for i in s:
            u.Insert_End(i)
        return u

l1 = LinkedList()
l2 = LinkedList()

while True:
    print("1.Insert List1")
