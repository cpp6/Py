class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def Insert_End(self, data):
        temp = Node(data)
        if self.head is None:
            self.head = temp
        else:
            q = self.head
            while q.next:
                q = q.next
            q.next = temp

    def Display(self):
        if self.head is None:
            print("Linked List is Empty")
        else:
            q = self.head
            while q:
                print(q.data, end="->")
                q = q.next
            print("None")

    def Sort(self):
        if self.head is None:
            print("List is Empty")
            return
        values = []
        q = self.head
        while q:
            values.append(q.data)
            q = q.next
        values.sort()
        print("Sorted List=", values)

l1 = LinkedList()

while True:
    print("1.Insert_End")
    print("2.Display")
    print("3.Sort in Ascending order")
    print("4.Exit")

    ch = int(input("Enter your choice: "))
    if ch == 1:
        data = int(input("Enter the Element: "))
        l1.Insert_End(data)
    elif ch == 2:
        l1.Display()
    elif ch == 3:
        l1.Sort()
    else:
        exit(0)
