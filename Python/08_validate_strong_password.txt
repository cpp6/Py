password = input("Enter password: ")

upper = lower = digit = special = False
special_chars = "@#$%&*!"

if len(password) >= 8:
    for ch in password:
        if ch.isupper():
            upper = True
        elif ch.islower():
            lower = True
        elif ch.isdigit():
            digit = True
        elif ch in special_chars:
            special = True

    if upper and lower and digit and special:
        print("Strong Password")
    else:
        print("Weak Password")
else:
    print("Password must be at least 8 characters long")
